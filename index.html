        // --- 紅蓮の花びらアニメーション ---

        const canvas = document.getElementById('fireCanvas');

        const ctx = canvas.getContext('2d');

        let petals = [];

        function resizeCanvas() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }

        window.addEventListener('resize', resizeCanvas);

        resizeCanvas();


        class Petal {

            constructor() { this.init(); }

            init() {

                this.x = Math.random() * canvas.width;

                this.y = canvas.height + Math.random() * 200;

                this.size = Math.random() * 15 + 8; 

                this.speedX = (Math.random() - 0.5) * 3;

                this.speedY = -(Math.random() * 4 + 2);

                this.alpha = 1; this.fade = Math.random() * 0.003 + 0.002;

                this.angle = Math.random() * Math.PI * 2;

                this.spin = (Math.random() - 0.5) * 0.05;

                this.flip = 0; this.flipSpeed = Math.random() * 0.03 + 0.01;

            }

            update() {

                this.x += this.speedX + Math.sin(this.y * 0.005) * 1.5;

                this.y += this.speedY; this.angle += this.spin; this.flip += this.flipSpeed; this.alpha -= this.fade;

                if (this.alpha <= 0 || this.y < -150) { this.init(); }

            }

            draw() {

                const r = 255; const g = Math.floor(Math.random() * 20); const b = 0;

                ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.angle);

                ctx.scale(Math.cos(this.flip) * this.size, this.size);

                ctx.globalCompositeOperation = 'lighter';

                ctx.beginPath(); ctx.moveTo(0, 0); ctx.quadraticCurveTo(0.5, -1, 1, 0); ctx.quadraticCurveTo(0.5, 1, 0, 0); ctx.closePath();

                const flicker = Math.random() * 0.3 + 0.7; ctx.fillStyle = `rgba(${r}, ${g}, ${b}, ${this.alpha * flicker})`; ctx.fill();

                ctx.restore();

                ctx.globalCompositeOperation = 'lighter';

                const grad = ctx.createRadialGradient(this.x, this.y, 0, this.x, this.y, this.size * 5);

                grad.addColorStop(0, `rgba(255, 0, 0, ${this.alpha * 0.15})`); grad.addColorStop(1, `rgba(255, 0, 0, 0)`);

                ctx.fillStyle = grad; ctx.beginPath(); ctx.arc(this.x, this.y, this.size * 5, 0, Math.PI * 2); ctx.fill();

                ctx.globalCompositeOperation = 'source-over';

            }

        }

        function initPetals() { petals = []; for (let i = 0; i < 80; i++) { petals.push(new Petal()); } }

        function animate() { ctx.clearRect(0, 0, canvas.width, canvas.height); petals.forEach(p => { p.update(); p.draw(); }); requestAnimationFrame(animate); }

        initPetals(); animate();
